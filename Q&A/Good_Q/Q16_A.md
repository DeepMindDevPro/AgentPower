def record_execution(self, record: ToolExecutionRecord):
    tool_name = record.tool_name
    
    # ① 创建或获取该工具的统计对象
    if tool_name not in self._stats:

    stat = self._stats[tool_name]
    
    # ② 更新统计指标
    stat.call_count += 1
    stat.total_time += record.execution_time
    stat.min_time = min(stat.min_time, record.execution_time)
    stat.max_time = max(stat.max_time, record.execution_time)
    
    # ③ 更新成功/失败计数
    if record.success:
        stat.successful_calls += 1
    else:
        stat.failed_calls += 1
    
    # ④ 保存执行记录
    self._records.append(record)
    
    # ⑤ 自动保存（如果启用）
    if self._auto_save and self._save_path:
        self.save_statistics()
```

| `min_time` | `min(current, new)` | 最小耗时 |
| `max_time` | `max(current, new)` | 最大耗时 |
| `success_rate` | `successful_calls / call_count * 100` | 成功率百分比 |

        result = await self._tool.execute(*args, **kwargs)  # ← 返回到此处
        
        # ┌────────────────────────────────────┐
        # │ 工具执行完成，现在计算统计           │
        # │ 堆栈：[0:Agent, 1:Wrapper]          │
        # └────────────────────────────────────┘
        
        # 计算执行耗时
        record.execution_time = time.perf_counter() - start_time
        #    ↑ 从 0.0 更新为实际耗时 (秒)
        
        # 记录成功状态
        record.success = result.success
        #    ↑ 从 False 更新为实际结果
        
        if result.error:
            record.error_message = result.error
            # ↑ 如果有错误消息，记录到 record
        
        # ┌────────────────────────────────────┐
        # │ 调用追踪器 → 堆栈扩展为 [0:Agent]   │
        # │                      [1:Wrapper]    │
        # │                      [2:Tracker]    │
        # └────────────────────────────────────┘
        self._tracker.record_execution(record)
        # ← 进入 Level 4
        
        # 返回结果给调用者
        return result
        #    ↑ 返回给 Level 0
        
    except Exception as e:
        # 异常路径
        execution_time = time.perf_counter() - start_time
        record.execution_time = execution_time
        record.error_message = f"{type(e).__name__}: {str(e)}"
        
        self._tracker.record_execution(record)
        raise
```

# 位置：mini_agent/statistics_wrapper.py:148-175
def record_execution(self, record: ToolExecutionRecord):
    """
    核心统计逻辑：更新统计数据
    堆栈状态：[0:Agent, 1:Wrapper, 2:Tracker]
    """
    tool_name = record.tool_name  # ← "read_file"
    
    # ┌────────────────────────────────────┐
    # │ 获取或创建该工具的统计对象           │
    # └────────────────────────────────────┘
    if tool_name not in self._stats:
        # 首次调用，创建新统计对象
        self._stats[tool_name] = ToolStat(name=tool_name)
        #   self._stats = {"read_file": ToolStat(...)}
    
    stat = self._stats[tool_name]  # ← 获取统计对象
    
    # ┌────────────────────────────────────┐
    # │ 更新统计指标                        │
    # └────────────────────────────────────┘
    stat.call_count += 1
    #   调用次数累加
    
    stat.total_time += record.execution_time
    #   累计总时间
    
    stat.min_time = min(stat.min_time, record.execution_time)
    #   更新最小耗时
    
    stat.max_time = max(stat.max_time, record.execution_time)
    #   更新最大耗时
    
    if record.success:
        stat.successful_calls += 1
    else:
        stat.failed_calls += 1
    #   更新成功/失败计数
    
    # ┌────────────────────────────────────┐
    # │ 保存执行记录                        │
    # └────────────────────────────────────┘
    self._records.append(record)
    #   self._records 列表追加记录对象
    
    # ┌────────────────────────────────────┐
    # │ 自动保存检查                        │
    # └────────────────────────────────────┘
    if self._auto_save and self._save_path:
        self.save_statistics()
        #   可能触发 Level 5: save_statistics()
        #   写入 JSON 文件

### 七、总结：堆栈级别执行流程

1. **调用入口**：Agent 主循环调用 `wrapped_tool.execute()`
2. **统计开始**：`StatisticsWrapper.execute()` 记录时间戳
3. **实际执行**：调用原始工具的 `execute()` 方法
4. **统计完成**：计算耗时，创建执行记录
5. **数据持久化**：调用 `tracker.record_execution()` 更新统计数据
6. **返回结果**：返回 `ToolResult` 给 Agent

整个过程通过**装饰器模式**实现，原始工具完全不知道被统计，实现了**无侵入式监控**。
